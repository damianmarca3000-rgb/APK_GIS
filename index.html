async function descargarData() { // 1. Convertimos los datos a formato JSON var data = JSON.stringify(drawnItems.toGeoJSON()); var blob = new Blob([data], {type: "application/json"}); var file = new File([blob], "inspeccion_campo.geojson", {type: "application/json"}); // 2. Verificamos si el navegador permite compartir archivos if (navigator.canShare && navigator.canShare({ files: [file] })) { try { await navigator.share({ files: [file], title: 'Datos de Supervisión', text: 'Aquí tienes los datos de inspección de campo.' }); } catch (error) { console.error('Error al compartir:', error); // Si falla, hacemos la descarga normal descargarArchivoNormal(blob); } } else { // Si el navegador no permite compartir archivos, hacemos descarga normal descargarArchivoNormal(blob); } } function descargarArchivoNormal(blob) { var a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = "inspeccion_campo.geojson"; a.click(); }
